A post demonstrating different fenced code blocks, inline code, and language highlighting.

## JavaScript Example

```js
function greet(name) {
  return `Hello, ${name}!`;
}

console.log(greet("World"));
```

## TypeScript Example

```ts
interface User {
  id: string;
  name: string;
  email?: string;
}

const users: User[] = [
  { id: "1", name: "Alice" },
  { id: "2", name: "Bob", email: "bob@example.com" },
];

export function getNames(list: User[]): string[] {
  return list.map(u => u.name);
}
```

## Bash Script
